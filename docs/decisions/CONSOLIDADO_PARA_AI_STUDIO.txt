// 1. Estructura de Tabla de Usuarios y Selector de Roles (React / JSX)
// Componente para listar usuarios y permitir la edición de su rol.
// -----------------------------------------------------------------------

import React, { useState } from 'react';

const UsersTable = () => {
  const [users, setUsers] = useState([
    { id: 1, name: 'Carlos Flores', email: 'carlos.flores@example.com', role: 'Admin' },
    { id: 2, name: 'Sofia Rodriguez', email: 'sofia.r@example.com', role: 'Editor' },
    { id: 3, name: 'Luis Gomez', email: 'luis.gomez@example.com', role: 'Viewer' },
    { id: 4, name: 'Ana Torres', email: 'ana.torres@example.com', role: 'Editor' },
  ]);

  const handleRoleChange = (userId, newRole) => {
    setUsers(users.map(user => 
      user.id === userId ? { ...user, role: newRole } : user
    ));
    console.log(`Usuario ${userId} ahora tiene el rol: ${newRole} (simulado)`);
    // Aquí iría la llamada a la API para actualizar el rol en el backend.
  };

  const RoleSelector = ({ userId, currentRole }) => {
    return (
      <select 
        value={currentRole} 
        onChange={(e) => handleRoleChange(userId, e.target.value)}
        className="bg-industrial-gray-light text-white rounded-md p-2 border-none"
      >
        <option value="Admin">Admin</option>
        <option value="Editor">Editor</option>
        <option value="Viewer">Viewer</option>
      </select>
    );
  };

  return (
    <div className="p-8 bg-gray-900 industrial-dark rounded-lg">
      <h2 className="text-xl font-bold text-gray-200 mb-4">Panel de Administración de Usuarios</h2>
      <div className="overflow-x-auto">
        <table className="min-w-full text-left text-sm text-gray-300">
          <thead className="bg-industrial-gray-dark">
            <tr>
              <th scope="col" className="px-6 py-4 font-medium">Nombre</th>
              <th scope="col" className="px-6 py-4 font-medium">Email</th>
              <th scope="col" className="px-6 py-4 font-medium">Rol</th>
              <th scope="col" className="px-6 py-4 font-medium text-center">Acciones</th>
            </tr>
          </thead>
          <tbody>
            {users.map((user) => (
              <tr key={user.id} className="border-b border-industrial-gray">
                <td className="px-6 py-4">{user.name}</td>
                <td className="px-6 py-4">{user.email}</td>
                <td className="px-6 py-4">
                  <RoleSelector userId={user.id} currentRole={user.role} />
                </td>
                <td className="px-6 py-4 text-center">
                   <button className="text-accent-orange hover:underline">Eliminar</button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
};
